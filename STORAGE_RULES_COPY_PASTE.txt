rules_version = '2';

service firebase.storage {
  match /b/{bucket}/o {

    // This is the CRITICAL fix.
    // It specifically matches the path structure used by the media upload feature.
    // A user can write to their own folder, inside a specific media ID folder.
    // Updated to support nested HLS directories
    match /media/{userId}/{allPaths=**} {
      allow read: if true;
      allow write: if request.auth != null && request.auth.uid == userId;
    }

    // Secure rule for user profile images.
    match /users/{userId}/profileImage/{fileName} {
      allow read: if true;
      allow write: if request.auth != null && request.auth.uid == userId;
    }

    // Secure rule for general content images (deities, stories, etc.)
    // Allows any authenticated user to upload, which is necessary for the admin forms.
    match /content-images/{allPaths=**} {
      allow read: if true;
      allow write: if request.auth != null;
    }

    // Secure rule for product images in the marketplace.
    match /product-images/{allPaths=**} {
      allow read: if true;
      allow write: if request.auth != null;
    }

    // Secure rule for contest images
    // Allows any authenticated user to upload, which is necessary for the admin forms.
    match /contests/{allPaths=**} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    
    // Secure rule for post images and videos
    // Includes HLS streaming files (master.m3u8 and .ts segments)
    // HLS files are stored at: posts/{userId}/hls/{videoId}/master.m3u8
    // and posts/{userId}/hls/{videoId}/segment*.ts
    match /posts/{userId}/{allPaths=**} {
      allow read: if true;
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Explicit rule for HLS files in posts (for clarity, though covered by above rule)
    // This ensures master.m3u8 and .ts segment files are accessible
    match /posts/{userId}/hls/{videoId}/{fileName} {
      allow read: if true;
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Explicit rule for HLS files in media (for clarity, though covered by above rule)
    // This ensures master.m3u8 and .ts segment files are accessible
    match /media/{userId}/hls/{videoId}/{fileName} {
      allow read: if true;
      allow write: if request.auth != null && request.auth.uid == userId;
    }

    // Secure rule for manifestation images, videos, and results
    match /manifestations/{userId}/{allPaths=**} {
      allow read: if true;
      allow write: if request.auth != null && request.auth.uid == userId;
    }

    // Secure rule for temple renovation request images and videos
    match /renovation_requests/{allPaths=**} {
      allow read: if true;
      allow write: if request.auth != null;
    }

    // Secure rule for temple maintenance fund documents/images
    match /maintenance_funds/{allPaths=**} {
      allow read: if true;
      allow write: if request.auth != null;
    }

    // Default rule: Deny all other read/write operations to prevent unauthorized access.
    // This is a security best practice.
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
